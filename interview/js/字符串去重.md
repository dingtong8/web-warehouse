字符串去重的几种方式

----

## for 遍历

```js
function unique1(str) {
    var newStr = "";
    var flag;
    for (var i = 0; i < str.length; i++) {
        flag = 1;
        for (var j = 0; j < newStr.length; j++) {
            if (str[i] == newStr[j]) {
                flag = 0;
                break;
            }
        }
        if (flag) newStr += str[i];
    }
    return newStr;
} 
```


## indexOf

```js
function unique2(str) {
    var newStr = "";
    for (var i = 0; i < str.length; i++) {
        if (newStr.indexOf(str[i]) == -1) {
            newStr += str[i];
        }
    }
    return newStr;
}  
```

```js
function unique3(str) {
    var newStr = "";
    for (var i = 0; i < str.length; i++) {
        if (newStr.search(str[i]) == -1)
            newStr += str[i];

    }
    return newStr;
}  
```


## 利用对象属性

```js
function unique4(str) {
    var obj = {};
    var newStr = "";
    for (var i = 0; i < str.length; i++) {
        if (!obj[str[i]]) {
            newStr += str[i];
            obj[str[i]] = 1;
        }
    }
    return newStr;
} 
```
