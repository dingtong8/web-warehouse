## 基本组件

绑定普通文本

```js
import { Component } from '@angular/core';

@Component({
  selector: 'my-app',
  template: `<h1>hello {{name}}<h1>`
})

export class AppComponent {
  name = 'Angular'
}
```

绑定对象属性

```js
import { Component } from '@angular/core';

@Component({
  selector: 'my-app',
  template: `<h1>hello {{obj.name}}, the version is {{obj.version}}<h1>`
})

export class AppComponent {
  obj = {
    name: 'Angular',
    version: '4.x'
  }
}
```
