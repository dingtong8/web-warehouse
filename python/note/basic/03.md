## 判断字符串是否为数字

```python
def isNumber(s):
  try:
    float(s)
    return True
  except ValueError:
    pass

  try:
    import unicodedata
    unicodedata.numeric(s)
    return True
  except (TypeError, ValueError):
    pass
  
  return False

print(isNumber('foo'))  # False
print(isNumber('1'))    # True
print(isNumber('1.5'))  # True
print(isNumber('-2'))   # True
print(isNumber('1e3'))  # True
```

## 斐波那契数列

```python
# 方法一
def nterms(num):
  n1 = 0
  n2 = 1
  count = 2
  print(n1, ',', n2, end = ',')
  while(count < int(num)):
    nth = n1 + n2
    print(nth, end = ',')
    n1 = n2
    n2 = nth
    count += 1

nterms(10)


# 方法二
def recur_fibo(n):
  if n <= 1:
    return n
  else:
    return (recur_fibo(n - 1) + recur_fibo(n - 2))

nterms = int(input())

if (nterms <= 0):
  print('输入正数')
else:
  for i in range(nterms):
    print(recur_fibo(i))
```




## 最大公约数

```python
# 方法一
def hcf(x, y):
  if x > y:
    smaller = y
  else:
    smaller = x

  for i in range(1, smaller + 1):
    if ((x % i == 0) and (y % i == 0)):
      hcf = i
  print(x, '和', y, '的最大公约数为：', hcf)

# 方法二
def hcf2(x, y):
  for i in range(1, min(x, y) + 1):
    if (y % i == 0) & (x % i == 0):
      hcf = i
  print(x, '和', y, '的最大公约数为：', hcf)

hcf2(int(54), int(24))
```



## 最小公倍数

```python
# 方法一
def lcm(x, y):
  if x > y:
    greater = x
  else:
    greater = y

  while(True):
    if ((greater % x == 0) and (greater % y == 0)):
      lcm = greater
      break
    greater += 1

  return lcm

# 方法二
def lcm2(x, y):
  s = x * y
  while y: x, y = y, x % y
  return s // x

print(lcm(120, 123))
print(lcm2(120, 123))
```
